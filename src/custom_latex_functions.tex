% Command to insert vertical lines and new pages
\newcommand{\DrawVerticalLines}[1]{%
    \par\bigskip
    \textbf{Inserting #1 empty pages}\par\medskip  % <-- added line

    \DrawVerticalLinesAux{#1}%
}

\newcommand{\DrawVerticalLinesAux}[1]{%
    \ifnum#1>0
        \begin{tikzpicture}[remember picture, overlay]
            \draw[thick] (1cm, 0cm) -- (1cm, -\textheight); % Vertical line
        \end{tikzpicture}
        \newpage
        \DrawVerticalLinesAux{\numexpr#1-1\relax}%
    \fi
}

\newcounter{quotectr}

\newenvironment{myquote}[1]
{%
	\refstepcounter{quotectr}%
	\renewcommand{\quoteauthor}{#1}%

}
